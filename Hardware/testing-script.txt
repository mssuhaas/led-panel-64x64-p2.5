#include <ArduinoJson.h>

#define MAX_DATA_SIZE 20

float values[MAX_DATA_SIZE];
char params[MAX_DATA_SIZE][10]; // Assuming max parameter name length is 10 characters
char units[MAX_DATA_SIZE][10];  // Assuming max unit name length is 10 characters
int dataCount = 0; // To keep track of the number of data points

void updateData(const char* jsonString) {
  DynamicJsonDocument doc(1024);
  deserializeJson(doc, jsonString);

  dataCount = doc["count"];

  JsonArray paramsArray = doc["params"];
  JsonArray valuesArray = doc["values"];
  JsonArray unitsArray = doc["units"];

  for (int i = 0; i < dataCount && i < MAX_DATA_SIZE; i++) {
    values[i] = valuesArray[i];
    strncpy(params[i], paramsArray[i], sizeof(params[i]) - 1);
    strncpy(units[i], unitsArray[i], sizeof(units[i]) - 1);
    params[i][sizeof(params[i]) - 1] = '\0';
    units[i][sizeof(units[i]) - 1] = '\0';
  }
}

void printData() {
  Serial.println("Stored Data:");
  for (int i = 0; i < dataCount; i++) {
    Serial.print("Parameter: ");
    Serial.print(params[i]);
    Serial.print("\tValue: ");
    Serial.print(values[i]);
    Serial.print("\tUnits: ");
    Serial.println(units[i]);
  }
}

void setup() {
  delay(2000);
  Serial.begin(115200);
  const char* apiResponse = "{\"count\": 5, \"params\": [\"PM2.5\", \"PM10\", \"Temp\", \" rH\", \"AQI\"], \"values\": [5.92, 19.82, 34.36, 69.55, 19.82], \"units\": [\"ug/m3\", \"ug/m3\", \"C\", \"%Rh\", \" \"], \"string\": \"PM2.5 : 5.92, PM10 : 19.82, Temperature : 34.36, Relative Humidity : 69.55, AQI : 19.82\"}";

  updateData(apiResponse);
  printData();
}
